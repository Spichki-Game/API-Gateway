* Game Scenes

** Get sessions

#+BEGIN_SRC restclient :var ~=entry-point
  GET :~/game-scenes
#+END_SRC


** Get active sessions

#+BEGIN_SRC restclient :var ~=entry-point
  GET :~/game-scenes?status=active
#+END_SRC


** Get wait sessions

#+BEGIN_SRC restclient :var ~=entry-point
  GET :~/game-scenes?status=wait
#+END_SRC


** Get number of sessions

#+BEGIN_SRC restclient :var ~=entry-point
  GET :~/game-scenes?agg=count
#+END_SRC



* Game Scene

** Create session

#+begin_src restclient :var ~=entry-point
  POST :~/game-scenes
       {"session_id": "12345",
           "players": ["SciBourne", "John", "Fetcher"]}
#+end_src


** Stop session

#+begin_src restclient :var ~=entry-point
  DELETE :~/game-scenes/12345
#+end_src


** Leave player

#+begin_src restclient :var ~=entry-point
  PATCH :~/game-scenes/12345/players
       {"leave": "John"}
#+end_src
